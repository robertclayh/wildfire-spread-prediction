\begin{frame}[plain]
\frametitle{Deep Learning Solution: Compact U-Net}
\begin{columns}[T,totalwidth=\textwidth]
  \begin{column}{0.58\textwidth}
    \textbf{\large Architecture Overview}
    \begin{itemize}
      \item \textbf{Model:} U-Net with ResNet-18 Encoder
      \item \textbf{Goal:} Capture spatial dependencies (e.g., fire jumping gaps).
      \item \textbf{Loss:} Weighted BCE + Dice (to handle imbalance).
    \end{itemize}
    \vspace{0.3cm}
    \textbf{\large Training Details}
    \begin{itemize}
      \item Inputs: Multi-channel tensor (Burn, Wind, Fuels...)
      \item Output: Next-day burn probability map.
    \end{itemize}
